# 宽度优先搜索

```c++
//树型
bool isComplete(int root)
{
	queue<int> que;
	que.push(root);
	while (!que.empty())
	{
		bool IsAllNull = true;
		bool FrontHaveNull = false;
		int LevelNum = que.size();
		while (LevelNum--)
		{
			int top = que.front(); que.pop();
			if (top != -1)
			{
				IsAllNull = false;
				if (FrontHaveNull)return false;
				last = top;
				que.push(tree[top].left);
				que.push(tree[top].right);
			}
			else
			{
				FrontHaveNull = true;
				que.push(-1);
				que.push(-1);
			}
		}
		if (IsAllNull)return true;
	}
}
```
